import React, { useMemo } from 'react'
import { useSpeechSynthesis } from "react-speech-kit";

function SpeackComponent(props) {
    const { speak } = useSpeechSynthesis();   
    
    function speaking(){
        speak({ text: `The players list we searched are around ${props.count} you can find the details here` })
    }

    // Memoize the component so it only re-renders when props.count changes
    const memoizedComponent = useMemo(() => (
        <div>
            {props.count ? speaking() : null}
        </div>
    ), [props.count]);

    return memoizedComponent;
}

export default SpeackComponent;
